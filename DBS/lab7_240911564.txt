DECLARE
    v_rno StudentTable.rollno%TYPE := &rollno; 
    v_gpa StudentTable.gpa%TYPE;
BEGIN
    SELECT gpa INTO v_gpa 
    FROM StudentTable 
    WHERE rollno = v_rno; 
    
    IF (v_gpa > 0 AND v_gpa <= 4) THEN
        DBMS_OUTPUT.PUT_LINE('F');
    ELSIF (v_gpa > 4 AND v_gpa <= 5) THEN
        DBMS_OUTPUT.PUT_LINE('E');
    ELSIF (v_gpa > 5 AND v_gpa <= 6) THEN
        DBMS_OUTPUT.PUT_LINE('D');
    ELSIF (v_gpa > 6 AND v_gpa <= 7) THEN
        DBMS_OUTPUT.PUT_LINE('C');
    ELSIF (v_gpa > 7 AND v_gpa <= 8) THEN
        DBMS_OUTPUT.PUT_LINE('B');
    ELSE
        DBMS_OUTPUT.PUT_LINE('A');
    END IF;
    
EXCEPTION
    WHEN NO_DATA_FOUND THEN 
        DBMS_OUTPUT.PUT_LINE('NO DATA FOUND FOR STUDENT W/ ROLL : ' || v_rno);
END;

DECLARE
    v_doi DATE;
    v_dor DATE;
    v_daysbwn NUMBER;
    v_fine NUMBER;  -- Added: store fine amount
BEGIN
    v_doi := TO_DATE('&issue_date', 'DD-MON-YYYY');
    v_dor := TO_DATE('&return_date', 'DD-MON-YYYY');
    
    v_daysbwn := v_dor - v_doi;
    
    IF(v_daysbwn <= 7) THEN
        DBMS_OUTPUT.PUT_LINE('NO FINE');
    ELSIF(v_daysbwn <= 15) THEN
        v_fine := 1*(v_daysbwn-7);
        DBMS_OUTPUT.PUT_LINE('FINE: ' || v_fine);
    ELSIF(v_daysbwn <= 30) THEN
        v_fine := 7+2*(v_daysbwn-15);
        DBMS_OUTPUT.PUT_LINE('FINE: ' || v_fine);
    ELSE
        v_fine := 37+(v_daysbwn-30)*5;
        DBMS_OUTPUT.PUT_LINE('FINE: ' || v_fine);
    END IF;
END;
/

Declare
	v_rollno_max  StudentTable.rollno%TYPE;
	v_gpa_max  StudentTable.GPA%TYPE;
	v_LetterGrade_max StudentTable.LetterGrade%TYPE;
BEGIN	
	SELECT gpa,rollno,LetterGrade into v_gpa_max,v_rollno_max,v_LetterGrade_max  from StudentTable
	where ROWNUM=1;
	FOR rec IN (SELECT RollNo, gpa,LetterGrade FROM StudentTable) LOOP
		if(rec.gpa>v_gpa_max) THEN	
			v_gpa_max:=rec.gpa;
			v_LetterGrade_max:=rec.LetterGrade;
			v_rollno_max:=rec.rollno;
		END IF;
	END LOOP;
    DBMS_OUTPUT.PUT_LINE('Student with MAX GPA:');
    DBMS_OUTPUT.PUT_LINE('RollNo: ' || v_rollno_max || ' GPA: ' || v_gpa_max);
END;
/

DECLARE
    v_instructor_id Instructor.ID%TYPE;
    v_name Instructor.name%TYPE;
    v_dept_name Instructor.dept_name%TYPE;
    v_salary Instructor.salary%TYPE;
    v_count NUMBER;
BEGIN
    -- Count matching instructors first
    SELECT COUNT(*) INTO v_count
    FROM Instructor 
    WHERE name = UPPER('&instructor_name');
    
    IF v_count = 0 THEN
        DBMS_OUTPUT.PUT_LINE('No instructor found with name: ' || UPPER('&instructor_name'));
    ELSIF v_count > 1 THEN
        DBMS_OUTPUT.PUT_LINE('Multiple (' || v_count || ') instructors found with name: ' || UPPER('&instructor_name'));
    ELSE
        -- Single instructor found - display details
        SELECT ID, name, dept_name, salary 
        INTO v_instructor_id, v_name, v_dept_name, v_salary
        FROM Instructor 
        WHERE name = UPPER('&instructor_name');
        
        DBMS_OUTPUT.PUT_LINE('Instructor Details:');
        DBMS_OUTPUT.PUT_LINE('ID: ' || v_instructor_id);
        DBMS_OUTPUT.PUT_LINE('Name: ' || v_name);
        DBMS_OUTPUT.PUT_LINE('Department: ' || v_dept_name);
        DBMS_OUTPUT.PUT_LINE('Salary: ' || v_salary);
    END IF;
    
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('No instructor found with name: ' || UPPER('&instructor_name'));
    WHEN TOO_MANY_ROWS THEN
        DBMS_OUTPUT.PUT_LINE('Multiple instructors found with name: ' || UPPER('&instructor_name'));
END;
/
