create table salary_raise(
	Instructor_Id NUMBER,
	Raise_date DATE,
	Raise_amt NUMBER(10,2)
	);
	

SET SERVEROUTPUT ON;
SET DEFINE OFF;
DECLARE
	v_dept_name VARCHAR(20) := 'Comp. Sci.';
	CURSOR c_inst IS
		SELECT id, salary 
		FROM Instructor 
		where dept_name=v_dept_name
		FOR UPDATE;
	v_id Instructor.id%type;
	v_old_sal Instructor.salary%TYPE;
	v_raise_amt Instructor.salary%type;
	v_raise_date DATE:=SYSDATE;

BEGIN
	--DBMS_OUTPUT.PUT_LINE('Before open : '||c_inst%ISOPEN);
	OPEN c_inst;
		--DBMS_OUTPUT.PUT_LINE('After opening : ' || c_inst%ISOPEN);
		LOOP
			FETCH c_inst INTO v_id,v_old_sal;
			v_raise_amt:=0.05*v_old_sal;
			UPDATE INSTRUCTOR set salary=1.05*salary where id=v_id;
			INSERT INTO salary_raise (Instructor_Id, Raise_date, Raise_amt)
    VALUES (v_id, v_raise_date, v_raise_amt);
			EXIT WHEN c_inst%NOTFOUND;
		END LOOP;
	CLOSE c_inst;
	COMMIT;
END;
/

SET SERVEROUTPUT ON;
DECLARE
	v_id STUDENT.id%type;
	v_name STUDENT.name%type;
	v_dept_name STUDENT.dept_name%type;
	v_tot_cred STUDENT.tot_cred%type;
	CURSOR c_inst IS 
		SELECT id,name,dept_name,tot_cred
		FROM STUDENT
		ORDER BY tot_cred ASC;
BEGIN
	OPEN c_inst;
	LOOP		
		FETCH c_inst INTO v_id,v_name,v_dept_name,v_tot_cred;
		DBMS_OUTPUT.PUT_LINE(v_id||v_name||v_dept_name||v_tot_cred);
		EXIT WHEN c_inst%ROWCOUNT>10 OR c_inst%NOTFOUND;
	END LOOP;
	CLOSE c_inst;
END;
/

SET SERVEROUTPUT ON;

DECLARE
  CURSOR c_course IS
    SELECT
      c.course_id,
      c.title,
      c.dept_name,
      c.credits,
      s.building,
      s.room_number,
      s.time_slot_id,
      i.name AS instructor_name
    FROM Course c
         JOIN Section s
              ON c.course_id = s.course_id
         JOIN Teaches tch
              ON s.course_id = tch.course_id
             AND s.section_id = tch.section_id
             AND s.semester = tch.semester
             AND s.year = tch.year
         JOIN Instructor i
              ON tch.id = i.id;

BEGIN
  FOR r IN c_course LOOP
    DBMS_OUTPUT.PUT_LINE(
      'Course_id=' || r.course_id ||
      ', Title=' || r.title ||
      ', Dept=' || r.dept_name ||
      ', Credits=' || r.credits ||
      ', Instructor=' || r.instructor_name ||
      ', Building=' || r.building ||
      ', Room=' || r.room_number ||
      ', Time=' || r.time_slot_id ||
      ', Tot_Students=' || 0
    );
  END LOOP;
END;
/

      SELECT s.reg_no
      FROM Student s
      JOIN Takes t ON s.reg_no = t.reg_no
      WHERE t.course_id = 'CS101'
        AND s.tot_cred < 30;

   FOR r IN c_students LOOP
      DELETE FROM Takes
      WHERE reg_no = r.reg_no
        AND course_id = 'CS101';
   END LOOP;

   COMMIT;  
END;

UPDATE StudentTable
SET LetterGrade = 'F';

DECLARE
   CURSOR c_student IS
      SELECT StudentID, GPA
      FROM StudentTable
      FOR UPDATE; 

   v_grade CHAR(1);
BEGIN
   FOR r IN c_student LOOP
      -- map GPA to letter grade
      IF r.GPA >= 3.7 THEN
         v_grade := 'A';
      ELSIF r.GPA >= 3.3 THEN
         v_grade := 'B+';
      ELSIF r.GPA >= 3.0 THEN
         v_grade := 'B';
      ELSIF r.GPA >= 2.7 THEN
         v_grade := 'C+';
      ELSIF r.GPA >= 2.3 THEN
         v_grade := 'C';
      ELSIF r.GPA >= 2.0 THEN
         v_grade := 'D';
      ELSE
         v_grade := 'F';
      END IF;

      UPDATE StudentTable
      SET LetterGrade = v_grade
      WHERE StudentID = r.StudentID;
   END LOOP;
END;
/

DECLARE
   v_course_id   Course.CourseID%TYPE := '&course_id';  

   CURSOR c_instructor IS
      SELECT i.Name
      FROM Instructor i
      JOIN Teaches t ON i.InstructorID = t.InstructorID
      WHERE t.CourseID = v_course_id;

BEGIN
   DBMS_OUTPUT.PUT_LINE('Instructors teaching course ' || v_course_id || ':');
   
   FOR r IN c_instructor LOOP
      DBMS_OUTPUT.PUT_LINE(r.Name);
   END LOOP;

   IF c_instructor%NOTFOUND THEN
      DBMS_OUTPUT.PUT_LINE('No instructor found teaching this course.');
   END IF;
   
END;
/

DECLARE
   CURSOR c_student IS
      SELECT DISTINCT s.StudentID, s.Name
      FROM Student s
      JOIN Takes t        ON s.StudentID = t.StudentID
      JOIN Teaches te     ON t.CourseID  = te.CourseID
      WHERE s.AdvisorID = te.InstructorID;  
BEGIN
   DBMS_OUTPUT.PUT_LINE('Students registered for a course taught by their advisor:');
   
   FOR r IN c_student LOOP
      DBMS_OUTPUT.PUT_LINE(r.StudentID || ' - ' || r.Name);
   END LOOP;

   IF c_student%NOTFOUND THEN
      DBMS_OUTPUT.PUT_LINE('No such students found.');
   END IF;
END;
/
