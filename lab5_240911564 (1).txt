1b)
select e.fname,e.address 
from employee e
join department d
on e.dno=d.dnumber
where d.dname='Research';

2)
select p.pnumber,p.dnum,e.lname,e.address,e.bdate
from project p
join employee e
on p.dnum=e.dno
join department d
on e.super_ssn=d.mgr_ssn
where p.plocation='Stafford';

3)
select e1.fname,e1.lname,e2.fname,e2.lname
from employee e1
join employee e2
on e1.ssn=e2.super_ssn;

4)
select p.pnumber
from project p
join employee e
on p.dnum=e.dno
and e.lname='Smith';

5)
update employee
set salary=salary*1.1
where exists(
	select 1 from project
	where pname='ProductX'
	and project.dnum=employee.dno
	);

select e.fname,d.dependent_name
from employee e 
join dependent d
on e.ssn=d.essn
and e.sex=d.sex;

8)
select fname
from employee e
where not exists(
	select 1 from dependent d
	where e.ssn=d.essn
	)
	
9)
select e.fname,d.mgr_ssn
from employee e
join department d
on e.ssn=d.mgr_ssn
where exists(
	select 1 from dependent
	where dependent.essn=e.ssn);
	
10)
select sum(salary),max(salary),min(salary),avg(salary)
from employee;

11)
select p.pnumber,p.pname,count(distinct(e.ssn))
from project p
join employee e
on p.dnum=e.dno
group by p.pnumber,p.pname;


12)
select p.pnumber,p.pname,count(distinct(e.ssn))
from project p
join employee e
on p.dnum=e.dno
group by p.pnumber,p.pname
having count(distinct(e.ssn))>2;

13)
select d.dname,d.dnumber,count(case when e.salary>40000 then 1 end) as high_pay_emp
from department d
join employee e
on d.dnumber=e.dno
group by d.dname,d.dnumber
having count(distinct(e.ssn))>5;
