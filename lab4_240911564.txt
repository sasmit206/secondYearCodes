11)
SELECT i.name
FROM instructor i
WHERE NOT EXISTS (
    SELECT s.id 
    FROM student s
    WHERE s.dept_name = 'Comp. Sci.'
    AND NOT EXISTS (
        SELECT 1
        FROM teaches t
        JOIN takes tk ON t.course_id = tk.course_id 
                      AND t.sec_id = tk.sec_id 
                      AND t.semester = tk.semester 
                      AND t.year = tk.year
        WHERE t.id = i.id AND tk.id = s.id
    )
);
